	function doViewAndCamo(affix) {
		//visibility
		var view = parseFloat($("#basicview"+affix).val());
		if (!isNaN(view)) {
			var bonusAdded=0;
			if ($('#vents'+affix).is(':checked')) {
				bonusAdded=bonusAdded+5;
			}
			if ($('#brother'+affix).is(':checked')) {
				bonusAdded=bonusAdded+5;
			}
			if ($('#consumables'+affix).is(':checked')) {
				bonusAdded=bonusAdded+10;
			}
			
			view = view/0.875*(0.00375*(100+bonusAdded)+0.5);
			
			var recon=parseFloat($("#reconskill"+affix).val());
			if (recon>0) {
				var added=0;
				if ($('#vents'+affix).is(':checked')) {
					added=added+5;
				}
				if ($('#brother'+affix).is(':checked')) {
					added=added+5;
				}
				if ($('#consumables'+affix).is(':checked')) {
					added=added+10;
				}
				recon=recon*(1+added/100);
	    		var reconSkill = (0.0002*recon+1);
	    		if (!isNaN(reconSkill)){
	    			view=view*reconSkill;
	    		}
			}
			
			var awarness=parseFloat($("#awarnessskill"+affix).val());
			if (awarness>0) {
				var added=0;
				if ($('#vents'+affix).is(':checked')) {
					added=added+5;
				}
				if ($('#brother'+affix).is(':checked')) {
					added=added+5;
				}
				if ($('#consumables'+affix).is(':checked')) {
					added=added+10;
				}
				awarness=(awarness+10)*(1+added/100);
	    		var awarnessSkill = (0.0003*awarness+1);
	    		if (!isNaN(awarnessSkill)){
	    			view=view*awarnessSkill;
	    		}
			}
			
			if($('#binocular'+affix).is(':checked')) {
				view = view * 1.25;
			} else if($('#optics'+affix).is(':checked')) {
				view = view * 1.1;
			}
			
			$("#view"+affix).text((view).toFixed(2));
		}
